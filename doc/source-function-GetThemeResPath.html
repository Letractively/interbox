 127: function GetTplPath($filename, $classname = NULL, $themeid = NULL, $lang = NULL) {
 128: 
 129:     if (empty($themeid))
 130:         $themeid = GetThemeID();
 131:     $themeid = intval($themeid);
 132: 
 133:     //get system path
 134:     $syspath = '';
 135:     if (defined('IBC1_SYSTEM_ROOT'))
 136:         $syspath = constant('IBC1_SYSTEM_ROOT');
 137: 
 138:     if (!empty($classname))
 139:         $classname.='/';
 140:     else
 141:         $classname = '';
 142: 
 143:     // get language code
 144:     if ($lang == NULL)
 145:         $lang = GetLang();
 146:     else
 147:         $lang = strtolower($lang);
 148: 
 149:     $theme = strval($themeid);
 150:     while (TRUE) {
 151:         if ($theme == '0') {
 152:             // system-level
 153:             $sysrespath = "{$syspath}templates/{$classname}";
 154:             $tplpath = $sysrespath . $lang . '/' . $filename;
 155:             if (is_file($tplpath))
 156:                 return $tplpath;
 157: 
 158:             // a language-neutral file
 159:             $tplpath = $sysrespath . $filename;
 160:             if (is_file($tplpath))
 161:                 return $tplpath;
 162: 
 163:             // not fully supported, needs aid from the default language
 164:             $dlang = strtolower(constant('IBC1_DEFAULT_LANGUAGE'));
 165:             if ($lang != $dlang) {
 166:                 if ($theme != '0') {
 167:                     // theme-level
 168:                     $themerespath = "{$syspath}themes/{$theme}/templates/{$classname}";
 169:                     $tplpath = $themerespath . $dlang . '/' . $filename;
 170:                     if (is_file($tplpath))
 171:                         return $tplpath;
 172:                 }
 173:                 // system-level
 174:                 $tplpath = $sysrespath . $dlang . '/' . $filename;
 175:                 if (is_file($tplpath))
 176:                     return $tplpath;
 177:             }
 178:             return '';
 179:         } else {
 180:             // theme-level
 181:             $themerespath = "{$syspath}themes/{$theme}/templates/{$classname}";
 182:             $tplpath = $themerespath . $lang . '/' . $filename;
 183:             if (is_file($tplpath))
 184:                 return $tplpath;
 185: 
 186:             // a language-neutral file
 187:             $tplpath = $themerespath . $filename;
 188:             if (is_file($tplpath))
 189:                 return $tplpath;
 190: 
 191:             $theme = '0';
 192:         }
 193:     }
 194: }

 258: /**
 259:  * gets language code according to http request and the system default setting.
 260:  *
 261:  * @return string
 262:  */
 263: function GetLang() {
 264:     $key = 'IBC1_Language';
 265: 
 266:     //not first time
 267:     if (isset($GLOBALS[$key]))
 268:         return $GLOBALS[$key];
 269: 
 270:     //get system path
 271:     $syspath = '';
 272:     if (defined('IBC1_SYSTEM_ROOT'))
 273:         $syspath = constant('IBC1_SYSTEM_ROOT');
 274: 
 275:     //by preference
 276:     $lang = strtolower(strCookie('Language'));
 277:     if ($lang != '' && is_dir("{$syspath}lang/{$lang}")) {
 278:         $GLOBALS[$key] = $lang;
 279:         return $lang;
 280:     }
 281: 
 282:     //by browser
 283:     $accepted = &$_SERVER['HTTP_ACCEPT_LANGUAGE'];
 284:     if (isset($accepted)) {
 285:         $l = explode(';', $accepted);
 286:         $l = explode(',', $l[0]);
 287:         foreach ($l as $lang) {
 288:             $lang = strtolower($lang);
 289:             if (is_dir("{$syspath}lang/{$lang}")) {
 290:                 $GLOBALS[$key] = $lang;
 291:                 return $lang;
 292:             }
 293:             $pos = strpos($lang, '-');
 294:             if ($pos > 0) {
 295:                 //e.g. zh-cn
 296:                 $lang = substr($lang, 0, $pos);
 297:                 if (is_dir("{$syspath}lang/{$lang}")) {
 298:                     $GLOBALS[$key] = $lang;
 299:                     return $lang;
 300:                 }
 301:             }
 302:         }
 303:     }
 304: 
 305:     //default
 306:     $GLOBALS[$key] = strtolower(constant('IBC1_DEFAULT_LANGUAGE'));
 307:     return $GLOBALS[$key];
 308: }

 588:  * remove all cached files

 590: function ClearCache() {
 591: 
 592:     //get system path
 593:     $syspath = '';
 594:     if (defined('IBC1_SYSTEM_ROOT'))
 595:         $syspath = constant('IBC1_SYSTEM_ROOT');
 596: 
 597:     $themes = array_keys(GetThemes());
 598:     foreach ($themes as $id) {
 599:         if (is_dir("{$syspath}cache/{$id}")) {
 600:             $languages = dir("{$syspath}cache/{$id}");
 601:             while (FALSE != ($lang = $languages->read())) {
 602:                 if (substr($lang, 0, 1) == '.') // omit '.','..','.xxx'
 603:                     continue;
 604:                 $files = dir("{$syspath}cache/{$id}/{$lang}");
 605:                 while (FALSE != ($file = $files->read())) {
 606:                     if (substr($file, 0, 1) == '.') // omit '.','..','.xxx'
 607:                         continue;
 608:                     unlink("{$syspath}cache/{$id}/{$lang}/{$file}");
 609:                 }
 610:                 $files->close();
 611:                 rmdir("{$syspath}cache/{$id}/{$lang}");
 612:             }
 613:             $languages->close();
 614:             rmdir("{$syspath}cache/{$id}");
 615:         }
 616:     }

 694:     final public function Route(array $config) {
 695: 
 696:         //default parameter names for module & function
 697:         $module = 'module';
 698:         $function = 'function';
 699: 
 700:         //custom parameter names for module & function
 701:         if (isset($config['definitions'])) {
 702:             $defconf = &$config['definitions'];
 703:             if (isset($defconf['module']))
 704:                 $module = $defconf['module'];
 705:             if (isset($defconf['function']))
 706:                 $function = $defconf['function'];
 707:         }
 708: 
 709:         //dynamically load and invoke functions
 710:         // either in or out of specified modules
 711:         $module_name = strGet($module);
 712:         $function_name = strGet($function);
 713:         if (!empty($module_name) && isset($config['modules'])) {
 714: 
 715:             //locate a module with a recursive approach
 716:             //?module=[module name]
 717:             $modconf = &$config['modules'];
 718:             if (isset($modconf[$module_name])) {
 719:                 return $this->Route($modconf[$module_name]);
 720:             }
 721:         } else if (!empty($function_name) && isset($config['functions'])) {
 722: 
 723:             //?function=[function name]
 724:             //?module=[module name]&function=[function name]
 725:             $funconf = &$config['functions'];
 726:             if (isset($funconf[$function_name])) {
 727: 
 728:                 //a process should only run once
 729:                 $this->CallProcess(
 730:                         $this->ConstructProcess($funconf[$function_name]), // construct an object of the Process
 731:                         $function_name, $module_name
 732:                 );
 733: 
 734:                 return TRUE;
 735:             }
 736:         }
 737: 
 738:         //a default view if there's no function invoked
 739:         if (isset($config['box'])) {
 740: 
 741:             //a single box view
 742:             $this->AddBox($this->ConstructBox($config['box']), NULL, $module_name);
 743: 
 744:             return TRUE;
 745:         } else if (isset($config['boxes'])) {
 746: 
 747:             //an array of boxes
 748:             foreach ($config['boxes'] as $f => $b)
 749:                 $this->AddBox($this->ConstructBox($b), is_string($f) ? $f : NULL, $module_name);
 750: 
 751:             return TRUE;
 752:         }
 753: 
 754:         // nothing found
 755:         return FALSE;
 756:     }
